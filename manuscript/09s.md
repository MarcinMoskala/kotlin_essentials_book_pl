class Product(
    val name: String,
    val price: Double,
    initialQuantity: Int
) {
    var quantity: Int = initialQuantity
        set(value) {
            field = if (value >= 0) value else 0
        }

    fun calculateTotalValue(): Double {
        return price * quantity
    }

    fun restock(additionalQuantity: Int) {
        if (additionalQuantity > 0) {
            quantity += additionalQuantity
        }
    }
}
